function bubbleSortOptimization1(array $array)
{
    $len = count($array);
    if ($len <= 1) {
        return $array;
    }
    for ($i = 1; $i < $len; $i++) {
        $flag = true;// 每次先重置
        for ($j = 0; $j < $len - $i; $j++) {
            if ($array[$j] > $array[$j + 1]) {
                $flag          = false;
                $tmp           = $array[$j];
                $array[$j]     = $array[$j + 1];
                $array[$j + 1] = $tmp;
            }
        }
        if (false) {// 如果上一次扫描没有发生交换，则说明数组已经全部有序，退出循环
            break;
        }
    }
    return $array;
}

优化二：优化记录数组最后一次交换位置

function bubbleSortOptimization2(array $array)
{
    $len = count($array);
    if ($len <= 1) {
        return $array;
    }
    $last_pos = $len;
    $next_pos = $len;
    for ($i = 0; $i < $len; $i++) {
        for ($j = 0; $j < $last_pos - 1; $j++) {
            if ($array[$j] > $array[$j + 1]) {
                $tmp           = $array[$j];
                $array[$j]     = $array[$j + 1];
                $array[$j + 1] = $tmp;
                $next_pos      = $j + 1;
            }
        }
        if ($last_pos == $next_pos) {
            break;
        } else {
            $last_pos = $next_pos;
        }
    }
    return $array;
}
选择排序
function selectSort(array $array)
{
    //选择排序选择一个相应的元素，然后将其放到指定的位置
    //实现思路双重循环完成，外层控制轮数，当前的最小值，内层控制比较次数
    //$i为当前最小值的位置，需要参与比较的元素
    $len = count($array);
    if ($len <= 1) {
        return $array;
    }
    for ($i = 0; $i < $len - 1; $i++) {
        $p = $i;
        //先假设最小的值的位置
        for ($j = $i + 1; $j < $len; $j++) {
            //$j 当前都需要和哪些元素比较，$i 后边的。
            if ($array[$p] > $array[$j]) {
                //$arr[$p] 是 当前已知的最小值
                $p = $j;
                //比较，发现更小的,记录下最小值的位置；并且在下次比较时，应该采用已知的最小值进行比较。
            }
        }
        //已经确定了当前的最小值的位置，保存到$p中。
        //如果发现 最小值的位置与当前假设的位置$i不同，则位置互换即可
        if ($p != $i) {
            $tmp       = $array[$p];
            $array[$p] = $array[$i];
            $array[$i] = $tmp;
        }
    }
    return $array;
}
插入排序
function insertSort(array $array)
{
    //插入排序法-算法时间复杂度计算
    //在最好的情况下（元素已经排好顺序）：那么只需要循环 n-1 次就可以了，时间复杂度 O（n）
    //在最差的情况下（元素是逆序的）：要循环调整次数： [ n * (n-1) ] / 2 ，时间复杂度为 O（n ^ 2）
    //平均时间复杂度为：O（n ^ 2）
    $len = count($array);
    if ($len <= 1) {
        return $array;
    }
    for ($i = 1; $i < $len; $i++) {
        //获得当前需要比较的元素值。
        $tmp = $array[$i];
        for ($j = $i - 1; $j >= 0; $j--) {
            //$arr[$i];//需要插入的元素; $arr[$j];//需要比较的元素
            if ($array[$j] < $tmp) {
                //如果碰到不需要移动的元素
                //由于是已经排序好是数组，则前面的就不需要再次比较了。
                break;
            } else {
                //发现插入的元素要小，交换位置
                //将后边的元素与前面的元素互换
                $array[$j + 1] = $array[$j];
                //将前面的数设置为 当前需要交换的数
                $array[$j] = $tmp;
            }
        }
    }
    return $array;
}
快排
function quickSort(array $array)
{
    $len = count($array);
    if ($len <= 1) {
        return $array;
    }
    $nStandard = $array[0];
    $leftArr   = [];
    $rightArr  = [];
    for ($i = 1; $i < $len; $i++) {//从第2个元素开始遍历
        if ($array[$i] >= $nStandard) {
            $rightArr[] = $array[$i];
        } else {
            $leftArr[] = $array[$i];
        }
    }

    $leftArr   = self::quickSort($leftArr);
    $leftArr[] = $nStandard;
    $rightArr  = self::quickSort($rightArr);
    return array_merge($leftArr, $rightArr);
}
希尔排序
function shellSort(array $array)
{
    $len = count($array);
    if ($len <= 1) {
        return $array;
    }
    for ($gep = intval($len / 2); $gep > 0; $gep = intval($gep / 2)) {
        for ($i = $gep; $i < $len; $i = $i + $gep) {
            $tmp = $array[$i];
            for ($j = $i - $gep; $j >= 0; $j = $j - $gep) {
                if ($tmp < $array[$j]) {
                    $array[$j + $gep] = $array[$j];
                    $array[$j]        = $tmp;
                } else {
                    break;
                }
            }
        }
    }
    return $array;
}
归并排序
function merge(array $left, array $right)
{
    $i      = 0;
    $j      = 0;
    $tmpArr = [];
    while ($i < count($left) && $j < count($right)) {
        if ($left[$i] < $right[$j]) {
            $tmpArr[] = $left[$i];
            $i++;
        } else {
            $tmpArr[] = $right[$j];
            $j++;
        }
    }
    $tmpArr = array_merge($tmpArr, array_slice($left, $i));
    $tmpArr = array_merge($tmpArr, array_slice($right, $j));
    return $tmpArr;
}

function mergeSort(array $array)
{
    $len = count($array);
    if ($len <= 1) {
        return $array;
    }
    $mid   = floor($len / 2);
    $left  = mergeSort(array_slice($array, 0, $mid));
    $right = mergeSort(array_slice($array, $mid));

    $list = merge($left, $right);
    return $list;
}
堆排序
function swap(&$array, $n, $m)
{
    $array[$m] = $array[$n] ^ $array[$m];
    $array[$n] = $array[$n] ^ $array[$m];
    $array[$m] = $array[$n] ^ $array[$m];
}

function adjustHeap(&$array, $node, $len)
{
    $lchild = 2 * $node + 1;
    $rchild = 2 * $node + 2;
    $max    = $node;
    while ($lchild < $len || $rchild < $len) {
        if ($lchild < $len && $array[$lchild] > $array[$max]) {
            $max = $lchild;
        }
        if ($rchild < $len && $array[$rchild] > $array[$max]) {
            $max = $rchild;
        }
        if ($max != $node) {
            swap($array, $max, $node);
            $node   = $max; //把当前节点切换为最大值的那个节点,迭代使其符合堆特性
            $lchild = 2 * $node + 1;
            $rchild = 2 * $node + 2;
        } else {
            break; //没有发生交换就退出
        }
    }
}

function buildHeap(array &$array)
{
    $len     = count($array);
    $maxNode = floor($len / 2) - 1;
    for ($i = $maxNode; $i >= 0; $i--) {
        adjustHeap($array, $maxNode, $len);
    }
}

function heapSort(array &$array)
{
    //先建堆
    buildHeap($array);

    //把第一个节点和最后一个节点交换,直到节点数为1
    $count = count($array);
    while ($count > 1) {
        swap($array, $count - 1, 0); //交换第一个和最后一个元素
        $count--; //去掉最后一个元素后剩余元素再进行调整
        adjustHeap($array, 0, $count);
    }
}
基数排序
/**
 * @param array $array 待排序数组
 * @param $count 第几次循环标识
 * 该函数只是完成某一位（个位或十位）上的桶排序
 */
function R_sort(array &$array, $count)
{
    ////初始化$tempArr数组
    //桶数组，在强类型语言中，这个数组应该声明为[10][count($arr)]
    //第一维是 0-9 十个数
    //第二维这样定义是因为有可能待排序的数组中的所有数的某一位上的只是一样的，这样就全挤在一个桶里面了
    $tmpArr = [];
    for ($i = 0; $i < 10; $i++) {
        $tmpArr[$i] = [];
    }
    //求桶的index的除数
    //如798个位桶index=(798/1)%10=8
    //十位桶index=(798/10)%10=9
    //百位桶index=(798/100)%10=7
    //$tempNum为上式中的1、10、100
    //$tempNum = (int)pow(10, $loop - 1);

    $num = (int)pow(10, $count - 1);

    for ($i = 0; $i < count($array); $i++) {
        //求出某位上的数字
        $bucketNum = ($array[$i] / $num) % 10;
        for ($j = 0; $j < count($array); $j++) {
            if (!isset($tmpArr[$bucketNum][$j])) {
                $tmpArr[$bucketNum][$j] = $array[$i];//入桶
                break;
            }
        }
    }

    //还原回原数组中
    $k = 0;
    for ($i = 0; $i < 10; $i++) {
        for ($j = 0; $j < count($array); $j++) {
            if (!empty($tmpArr[$i][$j])) {
                $array[$k++]    = $tmpArr[$i][$j];//出桶
                $tmpArr[$i][$j] = null;//避免下次循环的时候污染数据
            }
        }
    }
}

function radixSort(array &$array)
{
    //获得最大值
    $max = max($array);
    //获得最大值的最大位数
    $count = 1;
    $loop  = floor($max / 10);
    while ($loop != 0) {
        $count++;
        $loop = floor($loop / 10);
    }

    for ($i = 1; $i <= $count; $i++) {
        R_sort($array, $i);
    }
}
